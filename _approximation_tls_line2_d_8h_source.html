129  ElementType sx2 = ps.sx2() - ps.sx() * ps.sx();
130  ElementType sxy = ps.sxy() - ps.sx() * ps.sy();
131  ElementType sy2 = ps.sy2() - ps.sy() * ps.sy();
132 
133  ElementType trace_half = (sx2 + sy2) * 0.5;
134  ElementType eig_part = std::sqrt(trace_half * trace_half - sx2 * sy2 + sxy * sxy);
135 
136  sigma2 = trace_half - eig_part;
137 
138  ld.setX((sx2 - sigma2));
139  ld.setY((sxy));
140  ld.normalize();
141 
142  dist = ld.y() * ps.sx() - ld.x() * ps.sy();
143  return true;
144  }

163  ConstrainedType trim(const VectorType &beg, const VectorType &end) const
164  {
165  VectorType int_beg, int_end, n = normal();

195  static bool getCrossing(const ApproximationTlsLine2D<ElementType, ComputeType> &l1, const ApproximationTlsLine2D<ElementType, ComputeType> &l2, VectorType &crossing)
196  {
197  ElementType det = l1.a() * l2.b() - l1.b() * l2.a();
198  if (det == 0.0)
199  return false;
200  crossing.setX(-l1.c() * l2.b() + l2.c() * l1.b());
201  crossing.setY(l2.a() * l1.c()-l1.a() * l2.c());
202  crossing /= det;
203  return true;
204  }

212  static ElementType getErrorSquared(PrecSumsType ps)
213  {
214  ps.average();
