147  static DistanceType distance(const Transformation2D &tr1, const Transformation2D &tr2, DistanceType c_tr, DistanceType c_rot)
148  {
149  DistanceType a = std::atan2(tr1.int_rot(1, 0), tr1.int_rot(0, 0)) - std::atan2(tr2.int_rot(1, 0), tr2.int_rot(0, 0));
150  if (a > rtl::C_PI)
151  a -= 2.0 * rtl::C_PI;
152  else if (a <= -rtl::C_PI)
153  a += 2.0 * rtl::C_PI;
154  return a * a * c_rot + (tr1.int_translation - tr2.int_translation).lengthSquared() * c_tr;
155  }
