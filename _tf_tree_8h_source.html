259  TfChain<TransformationType> tf(const KeyType& from, const KeyType& to) const
260  {
261  const NodeType *from_node = &nodes.at(from);
262  const NodeType *to_node = &nodes.at(to);
263  std::list<TransformationType> ret;
264  size_t common_depth = std::min(from_node->depth(), to_node->depth());
265 
266  if (from_node->depth() > common_depth)
267  for (;from_node->depth() != common_depth; from_node = from_node->parent())
268  ret.push_back(from_node->tf().inverted());
269  auto insert_pos = ret.end();
270  if (to_node->depth() > common_depth)
271  for (; to_node->depth() != common_depth; to_node = to_node->parent())
272  insert_pos = ret.insert(insert_pos, to_node->tf());
273 
274  while (from_node->key() != to_node->key())
275  {
276  insert_pos = ret.insert(insert_pos, to_node->tf());
277  ret.insert(insert_pos, from_node->tf().inverted());
278  from_node = from_node->parent();
279  to_node = to_node->parent();
280  }
281 
282  return TfChain<TransformationType>(ret);
283  }
