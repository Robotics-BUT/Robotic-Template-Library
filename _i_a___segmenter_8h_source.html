138  void addPoint(const VectorType &pt, const VectorType &origin = VectorType::zeros())
139  {
140  bool has_cluster = false;
141  size_t cl_p = -1;
142 
143  // cluster pertinence search
144  for (auto it = points.begin(); it != points.end(); it++)
145  {
146  ElementType dist2 = scale_factor2 * VectorType::distanceSquared(pt, origin);
147  if (dist2 < l_bound2)
148  dist2 = l_bound2;
149  if (dist2 > u_bound2)
150  dist2 = u_bound2;
151 
152  if (VectorType::distanceSquared(pt, it->pt) < dist2)
153  {
154  if (!has_cluster)
155  {
156  has_cluster = true;
157  cl_p = it->pert;
158  }
159  else if (cl_p != it->pert)
160  {
161  for (auto merge_it = points.begin(); merge_it != points.end(); merge_it++)
162  {
163  if (merge_it->pert == cl_p)
164  {
165  merge_it->pert = it->pert;
166  }
167  }
168  cl_p = it->pert;
169  }
170  }
171  }
172 
173  if (!has_cluster)
174  {
175  cl_p = cluster_counter++;
176  }
177 
178  PointPert pp;
179  pp.pt = pt;
180  pp.pert = cl_p;
181  points.emplace_back(pp);
182  clusters_alive_refs[cl_p]++;
183 
184  if (points.size() > step_size)
185  {
186  auto to_cl = points.front();
187  points.pop_front();
188  clusters_alive[to_cl.pert].push_back(to_cl.pt);
189  clusters_alive_refs[to_cl.pert]--;
190 
191  if (clusters_alive_refs[to_cl.pert] == 0)
192  {
193  auto closed_cl = clusters_alive.extract(to_cl.pert);
194  clusters_closed.insert(std::move(closed_cl));
195  }
196  }
197  }
