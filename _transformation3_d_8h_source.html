153  Transformation3D<Element> operator()(const Transformation3D<Element> &tr) const
154  {
155  Transformation3D<Element> ret;
156  ret.int_rot = int_rot * tr.int_rot;

162  static DistanceType distance(const Transformation3D<Element> &tr1, const Transformation3D<Element> &tr2, Element c_tr, Element c_rot)
163  {
164  DistanceType a = tr1.int_aa.angle() - tr2.int_aa.angle();
165  if (a > rtl::C_PIf)
166  a -= 2.0f * rtl::C_PIf;
167  else if (a <= -rtl::C_PIf)
168  a += 2.0f * rtl::C_PIf;
169  return a * a * c_rot + (tr1.int_translation - tr2.int_translation).lengthSquared() * c_tr;
170  }
