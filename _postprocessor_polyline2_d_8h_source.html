78  bool operator()(const std::vector<VectorType> &pts, const std::vector<Approximation> &lines, const std::vector<IndexType> &indices)
79  {
80  if (lines.size() != indices.size() || lines.size() == 0)
81  return false;
82  int_polyline.clear();
83  int_polyline.emplace_back(lines.front().project(pts.front()));
84  for (size_t i = 1; i < lines.size(); i++)
85  {
86  if (!Approximation::getCrossing(lines[i - 1], lines[i], vec_tmp))
87  return false;
88  int_polyline.push_back(vec_tmp);
89  }
90  int_polyline.emplace_back(lines.back().project(pts.back()));
91  return true;
92  }
