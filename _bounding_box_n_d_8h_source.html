166  template<typename T>
167  std::vector<typename std::invoke_result_t<T, VectorType>> allVertices(T &&func = identityFunc) const
168  {
169  std::vector<typename std::invoke_result_t<T, VectorType>> vertices;
170  vertices.reserve(1u << dim);
171  VectorType v;
172  for (size_t i = 0; i < (1u << dim); i++)
173  {
174  for (size_t j = 0; j < dim; j++)
175  v[j] = (i & (1u << j)) ? b_min[j] : b_max[j];
176  vertices.push_back(func(v));
177  }
178  return vertices;
179  }

276  Element intersectionOverUnion(const BoundingBoxND &bb) const
277  {
278  auto intersected = BoundingBoxND<dim, ElementType>::intersection(*this, bb);
279  if(intersected)
280  {
281  Element intersection_vol = intersected->volume();
282  Element union_vol = volume() + bb.volume() - intersection_vol;
283  return intersection_vol / union_vol;
284  }
285  return 0.0;

295  static std::unique_ptr<BoundingBoxND<dim, ElementType>> intersection(const BoundingBoxND<dim, ElementType> &bb1, const BoundingBoxND<dim, ElementType> &bb2)
296  {
297  VectorType t_min(bb1.b_min.data().array().max(bb2.b_min.data().array()).matrix());
298  VectorType t_max(bb1.b_max.data().array().min(bb2.b_max.data().array()).matrix());
299  if ((t_min.data().array() < t_max.data().array()).all())
300  return std::make_unique<BoundingBoxND<dim, ElementType>>(t_min, t_max);
301  else
