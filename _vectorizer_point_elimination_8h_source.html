84  void operator()(std::vector<VectorType> & input, std::vector<LineSegmentType> & output)
85  {
86  break_pts.clear();
87  polyline.clear();
88  work_pt = 0;
89 
90  break_pts.push_back(input.size() - 1);
91 
92  while (!break_pts.empty())
93  {
94  if (break_pts.back() - work_pt < 2)
95  {
96  polyline.push_back(input[work_pt]);
97  work_pt = break_pts.back();
98  break_pts.pop_back();
99  }
100  else
101  {
102  max_dist = 0;
103  max_dist_index = 0;
104 
105  LineSegmentType ls(input[work_pt], input[break_pts.back()]);
106 
107  for (size_t i = work_pt + 1; i < break_pts.back(); i++)
108  {
109  d = ls.distanceToPointSquared(input[i]);
110  if (d > max_dist)
111  {
112  max_dist = d;
113  max_dist_index = i;
114  }
115  }
116 
117  if (max_dist > epsilon2)
118  break_pts.push_back(max_dist_index);
119  else
120  {
121  polyline.push_back(input[work_pt]);
122  work_pt = break_pts.back();
123  break_pts.pop_back();
124  }
125  }
126  }
127  polyline.push_back(input[work_pt]);
128 
129  output.clear();
130  for (size_t i = 1; i < polyline.size(); i++)
131  output.emplace_back(polyline[i - 1], polyline[i]);
132  }
133 
134  private:
135  ElementType d{}, max_dist{}, epsilon2{0.000001};
136  size_t work_pt{}, max_dist_index{};
137  std::vector<size_t > break_pts;

189  {
190  polyline.clear();
191  polyline.push_back(input[0]);
192  kp = 0;
193  wp = 1;
194  tp = 2;
195 
196  while (tp < input.size())
197  {
198  LineSegmentType ls(input[wp], input[kp]);
199  d = 0;
200  while (d < epsilon2 && tp < input.size())
201  {
202  d = ls.distanceToPointSquared(input[tp]);
203  tp++;
204  }
205 
206  if (tp == input.size())
207  {
208  polyline.push_back(input.back());
209  break;
210  }
211 
212  polyline.push_back(input[tp]);
213  kp = wp;
214  wp = tp;
215  tp++;
216  }
217 
218  output.clear();
219  for (size_t i = 1; i < polyline.size(); i++)
220  output.emplace_back(polyline[i - 1], polyline[i]);
221  }
