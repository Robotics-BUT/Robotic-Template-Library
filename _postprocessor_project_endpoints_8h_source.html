69  bool operator()(const std::vector<VectorType> &pts, const std::vector<Approximation> &approximations, const std::vector<IndexType> &indices)
70  {
71  if (approximations.size() != indices.size() || approximations.size() == 0)
72  return false;
73  int_output.clear();
74  int_output.reserve(approximations.size());
75  for (size_t i = 0; i < approximations.size(); i++)
76  int_output.emplace_back(approximations[i].trim(pts.begin() + indices[i].first, pts.begin() + indices[i].second));
77  return true;
78  }
