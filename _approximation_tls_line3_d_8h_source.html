94  bool operator()(PrecSumsType ps)
95  {
96  ps.average();
97  Eigen::Matrix<ComputeType, 3, 3> cov_m;
98  cov_m(0, 0) = ps.sx2() - ps.sx() * ps.sx();
99  cov_m(1, 0) = ps.sxy() - ps.sx() * ps.sy();
100  cov_m(2, 0) = ps.szx() - ps.sz() * ps.sx();
101  cov_m(1, 1) = ps.sy2() - ps.sy() * ps.sy();
102  cov_m(2, 1) = ps.syz() - ps.sy() * ps.sz();
103  cov_m(2, 2) = ps.sz2() - ps.sz() * ps.sz();
104 
105  Eigen::SelfAdjointEigenSolver<Eigen::Matrix<ComputeType, 3, 3>> solver;
106  solver.computeDirect(cov_m);
107  ld = VectorType(solver.eigenvectors().col(2).template cast<ElementType>());
108  lp = VectorType(ps.sx(), ps.sy(), ps.sz());
109  sigma2 = solver.eigenvalues()[0] + solver.eigenvalues()[1];
110 
111  return true;
112  }

131  ConstrainedType trim(const VectorType &beg, const VectorType &end) const
132  {
