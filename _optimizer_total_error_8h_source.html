101  auto totalError = [&sum_array, bp_cnt, bp_last_i](size_t *bp)
102  {
103  ElementType sigma2 = 0;
104  size_t sum_beg = 0;
105  for (size_t i = 0; i < bp_cnt; i++)
106  {
107  sigma2 += Approximation::getErrorSquared(sum_array.sums(sum_beg, bp[i]));
108  sum_beg = bp[i];
109  }
110  sigma2 += Approximation::getErrorSquared(sum_array.sums(sum_beg, bp_last_i));
111  return sigma2;
112  };

114  auto forwardHomogenize = [bp_last_i, bp_cnt](size_t *bp)
115  {
116  if (bp[0] < 2 || bp[0] > bp_last_i)
117  bp[0] = 2;
118  for (size_t i = 1; i < bp_cnt; i++)
119  if (bp[i] < bp[i - 1] + 2 || bp[i] > bp_last_i)
120  bp[i] = bp[i - 1] + 2;
121  };

123  auto wholeSimplexValidation = [bp_cnt, bp_array]()
124  {
125  size_t *array_end = bp_array + bp_cnt * (bp_cnt + 1);
126  for (size_t *p_checked = bp_array + bp_cnt; p_checked < array_end; p_checked += bp_cnt)
127  {
128  for (size_t *p_against = bp_array; p_against < p_checked; p_against += bp_cnt)
129  {
130  for (size_t i = 0; i < bp_cnt; i++)
131  if (p_checked[i] != p_against[i])
132  goto vertices_different;
133  return false;
134  vertices_different:;
135  }
136  }
137  return true;
138  };

204  bp_mean[i] = bp_sums[i] / bp_cnt; // compute the mean vector
205  tmp_pt = 2 * bp_mean[i] - discard[i]; // compute REFLECT vector

283  }

289  if(!newVertexValidation(bp_tmp))
290  break;
291  opt_vec_order.emplace(opt_err_tmp2, bp_tmp); // simplex CONTRACTed INSIDE
292  for (i = 0; i < bp_cnt; i++)
293  bp_sums[i] += bp_tmp[i];
294  bp_tmp = discard;
295  //std::cout<<"simplex CONTRACTed INSIDE"<<std::endl;
296  continue;
297  }
298  }
299  // simplex SHRINKed
300  //std::cout<<"simplex SHRINKed"<<std::endl;
301  // recompute all but the first vector
302  opt_err_tmp1 = opt_vec_order.begin()->first;
303  bp_first = opt_vec_order.begin()->second;
304  for (i = 0; i < bp_cnt; i++)

307  {
308  for (i = bp_cnt - 1; i < bp_cnt; i--)

311  if (i == bp_cnt - 1)

323  opt_vec_order.emplace(opt_err_tmp1, bp_first);

325  {
