132  void precompute(const std::vector<rtl::Vector2D<ElementType>> &vec)
133  {
134  size_t vec_size = vec.size();
135  BaseType::resize(vec_size);
136 
137  Eigen::Map<const Eigen::Matrix<ElementType, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor>> map(&vec[0][0], vec_size, 2);
138  BaseType::array.block(1, SumsType::cx, vec_size, 1) = map.array().col(SumsType::cx).template cast<ComputeType>();
139  BaseType::array.block(1, SumsType::cy, vec_size, 1) = map.array().col(SumsType::cy).template cast<ComputeType>();
140  BaseType::array.block(1, SumsType::cx2, vec_size, 1) = BaseType::array.block(1, SumsType::cx, vec_size, 1) * BaseType::array.block(1, SumsType::cx, vec_size, 1);
141  BaseType::array.block(1, SumsType::cy2, vec_size, 1) = BaseType::array.block(1, SumsType::cy, vec_size, 1) * BaseType::array.block(1, SumsType::cy, vec_size, 1);
142  BaseType::array.block(1, SumsType::cxy, vec_size, 1) = BaseType::array.block(1, SumsType::cx, vec_size, 1) * BaseType::array.block(1, SumsType::cy, vec_size, 1);
