93  bool operator()(PrecSumsType ps)
94  {
95  ps.average();
96  Eigen::Matrix<ComputeType, 3, 3> cov_m;
97  cov_m(0, 0) = ps.sx2() - ps.sx() * ps.sx();
98  cov_m(1, 0) = ps.sxy() - ps.sx() * ps.sy();
99  cov_m(2, 0) = ps.szx() - ps.sz() * ps.sx();
100  cov_m(1, 1) = ps.sy2() - ps.sy() * ps.sy();
101  cov_m(2, 1) = ps.syz() - ps.sy() * ps.sz();
102  cov_m(2, 2) = ps.sz2() - ps.sz() * ps.sz();
103 
104  Eigen::SelfAdjointEigenSolver<Eigen::Matrix<ComputeType, 3, 3>> solver;
105  solver.computeDirect(cov_m);
106  pn = VectorType(solver.eigenvectors().col(0).template cast<ElementType>());
107  pd = VectorType::scalarProjectionOnUnit(VectorType(ps.sx(), ps.sy(), ps.sz()), pn);
108  sigma2 = solver.eigenvalues()[0];
109 
110  return true;
111  }

136  {
137  static_assert(std::is_constructible<VectorType, typename Iter::value_type>::value, "Invalid iterator value_type.");
138  ConstrainedType out(pn, pd);
