81  void setRot(const VectorType &v1, const VectorType &v2)
82  {
83  VectorType v1_on = v1.normalized();
84  VectorType v2_on = (v2 - v1_on.dot(v2) * v1_on).normalized();
85  VectorType v2_n = v2.normalized();
86  ElementType cos_a = VectorType::scalarProjectionOnUnit(v2_n, v1_on);
87  ElementType sin_a = VectorType::scalarProjectionOnUnit(v2_n, v2_on);
88  int_rot_mat = MatrixType::identity() + sin_a * (v2_on.outer(v1_on) - v1_on.outer(v2_on)) + (cos_a - 1) * (v1_on.outer(v1_on) + v2_on.outer(v2_on));
89  childThis().rotMatUpdated();
90  }

321  void setRot(const VectorType &v1, const VectorType &v2)
322  {
323  VectorType v1_on = v1.normalized();
324  VectorType v2_on(-v1_on.y(), v1_on.x());
325  VectorType v2_n = v2.normalized();
326  ElementType cos_a = v1_on.dot(v2_n);
327  ElementType sin_a = v2_on.dot(v2_n);
328  this->int_rot_mat(0, 0) = cos_a;
329  this->int_rot_mat(0, 1) = -sin_a;
330  this->int_rot_mat(1, 0) = sin_a;
331  this->int_rot_mat(1, 1) = cos_a;
332  }
