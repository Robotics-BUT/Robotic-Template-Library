70  bool operator()(const std::vector<VectorType> &pts, std::vector<Approximation> &approximations, std::vector<IndexType> &indices)
71  {
72  beg_i = 0;
73  end_i = 0;
74  sums.setZero();
75 
76  approximations.clear();
77  indices.clear();
78 
79  while (end_i < pts.size())
80  {
81  PrecSumsType ps(pts[end_i].template cast<ComputeType>());
82  sums += ps;
83  if (Approximation::getErrorSquared(sums) < err2)
84  {
85  end_i++;
86  continue;
87  }
88 
89  approximations.emplace_back(sums - ps);
90  indices.emplace_back(beg_i, end_i);
91 
92  sums = ps;
93  beg_i = end_i;
94  end_i++;
95  }
96  if (end_i - beg_i >= 2)
97  {
98  approximations.emplace_back(sums);
99  indices.emplace_back(beg_i, end_i);
100  }
101  return true;
102  }
103 
104  private:
105  size_t beg_i{}, end_i{};
106  ElementType err2;
107  PrecSumsType sums;
108  };
