285  bool closestPoint(const LineSegmentND_common<dimensions, Element, ChildTemplate> &ls, typename VectorType::DistanceType &t) const
286  {

335  /* Calculation:
336  * vectors are denoted in capital, real numbers are small
337  * dot product is denoted using dot '.'
338  * n-dimensional line: B + lD = X, B = beginning, l = real parameter, D = unit direction, X = point on line
339  * n-dimensional hyperplane: N.X = c, N = normal unit vector of the hyperplane, X = point on it, c = distance from origin
340  * normal of the hyperplane always points into the examined area, therefore cosine of the angle between N and D demarcates if the line goes into te area (cos > 0) or leaves it (cos < 0)
341  * hyperplanes demarcating te examined area are always orthogonal to remaining axes, which means that dot product is reduced to single multiplication
342  * for each line/hyperplane v_tmp, the l parameter is identified
343  * the line crosses the examined area if and only if all l for line entrance are smaller than all l for line exit
344  */
345  bool cropByHyperRect(const VectorType &corner1, const VectorType &corner2, Element &segment_beg_t, Element &segment_end_t) const
346  {
347  Element in_max = std::numeric_limits<Element>::lowest(), out_min = std::numeric_limits<Element>::max();
348  Element l1, l2;
349 
350  for (size_t i = 0; i < dimensions; i++)
351  {
352  if (int_dir[i] != 0)
353  {
354  if (corner1[i] < corner2[i])
355  {
356  l1 = (corner1[i] - int_beg[i]) / int_dir[i];
357  l2 = (corner2[i] - int_beg[i]) / int_dir[i];
358  if (int_dir[i] > 0)
359  {
360  if (l1 > in_max) in_max = l1;
361  if (l2 < out_min) out_min = l2;
362  } else
363  {
364  if (l1 < out_min) out_min = l1;
365  if (l2 > in_max) in_max = l2;
366  }
367  } else
368  {
369  l1 = (corner1[i] - int_beg[i]) / int_dir[i];
370  l2 = (corner2[i] - int_beg[i]) / int_dir[i];
371  if (int_dir[i] > 0)
372  {
373  if (l1 < out_min) out_min = l1;
374  if (l2 > in_max) in_max = l2;
375  } else
376  {
377  if (l1 > in_max) in_max = l1;
378  if (l2 < out_min) out_min = l2;
379  }
380  }
381  }
382  }
383 
384  if (in_max > out_min)
385  return false;
386  else
387  {
388  segment_beg_t = in_max;
389  segment_end_t = out_min;
390  return true;
391  }
392  }

702  static bool getCrossing(const VectorType &first_beg, const VectorType &first_end, const VectorType &second_beg, const VectorType &second_end, Element &t_first, Element &t_second)
703  {
704  VectorType v_first = first_end - first_beg;
705  VectorType v_second = second_beg - second_end;
706  VectorType v_begs = second_beg - first_beg;
707 
708  Element det = v_first[0] * v_second[1] - v_second[0] * v_first[1];
709  t_first = v_begs[0] * v_second[1] - v_second[0] * v_begs[1];
710  t_second = v_first[0] * v_begs[1] - v_begs[0] * v_first[1];
711 
712  if (det == 0.0f)
713  {
714  if (t_first > 0.0f)
715  t_first = std::numeric_limits<ElementType>::max();
716  else if (t_first < 0.0f)
717  t_first = std::numeric_limits<ElementType>::lowest();
718 
719  if (t_second > 0.0f)
720  t_second = std::numeric_limits<ElementType>::max();
721  else if (t_second < 0.0f)
722  t_second = std::numeric_limits<ElementType>::lowest();
723 
724  return false;
725  }
726 
727  t_first /= det;
728  t_second /= det;
729  return true;
730  }

742  static bool getCrossing(const VectorType &first_beg, const VectorType &first_end, const VectorType &second_beg, const VectorType &second_end, VectorType &crossing)
743  {
744  VectorType v_first = first_end - first_beg;
745  VectorType v_second = second_beg - second_end;
746  VectorType v_begs = second_beg - first_beg;
747 
748  Element det = v_first[0] * v_second[1] - v_second[0] * v_first[1];
749  if (det == 0.0f)
750  return false;
751 
752  float t = (v_begs[0] * v_second[1] - v_second[0] * v_begs[1]) / det;
753  crossing = first_beg + t * v_first;
754  return true;
