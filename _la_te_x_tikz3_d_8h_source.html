723  above = std::move(mp);
724  return;
725  }
726 
727  if (proj_2d.distanceToPointSquared(mp->proj_2d) < std::pow(mp->radius * mp->scale, 2))
728  {
729  auto t_z = (mp->pos_3d.z() - ls_3d.beg().z()) / (ls_3d.end().z() - ls_3d.beg().z());
730  auto m_plane_intersection = -Vector2f(ls_3d.end().x() - ls_3d.beg().x(), ls_3d.end().y() - ls_3d.beg().y()) * t_z * fl / mp->pos_3d.z();
731  auto i_on_l = proj_2d.scalarProjection(m_plane_intersection);
732  auto m_on_l = proj_2d.scalarProjection(mp->proj_2d);
733  if ((ls_3d.direction().z() > 0) ^ (i_on_l > m_on_l))
734  under = std::move(mp);
735  else
736  above = std::move(mp);
737  }
738  else
739  under = std::move(mp); // "does not matter" is correct in principle, just used "under" in bipolar decision scheme
740  }

742  void splitSort(std::unique_ptr<LinePrimitive> lp, std::unique_ptr<LinePrimitive> &under, std::unique_ptr<LinePrimitive> &above, [[maybe_unused]]float fl, float eps) override
743  {
744  float t1, t2;
745  LineSegment2f::getCrossing(proj_2d, lp->proj_2d, t1, t2);
746  if (t1 > eps && t1 < 1.0 - eps && t2 > eps && t2 < 1 - eps)
747  {
748  Vector2f crossing(proj_2d.beg() + t1 * (proj_2d.end() - proj_2d.beg()));
749  float l1, l2;
750  t1 = crossing.x() * ls_3d.direction().z() - ls_3d.direction().x();
751  t2 = crossing.y() * ls_3d.direction().z() - ls_3d.direction().y();
752  if (std::abs(t1) > std::abs(t2))
753  l1 = (ls_3d.beg() + ls_3d.direction() * (ls_3d.beg().x() - crossing.x() * ls_3d.beg().z()) / t1).lengthSquared();
754  else
755  l1 = (ls_3d.beg() + ls_3d.direction() * (ls_3d.beg().y() - crossing.y() * ls_3d.beg().z()) / t2).lengthSquared();
756 
757  t1 = crossing.x() * lp->ls_3d.direction().z() - lp->ls_3d.direction().x();
758  t2 = crossing.y() * lp->ls_3d.direction().z() - lp->ls_3d.direction().y();
759  if (std::abs(t1) > std::abs(t2))
760  l2 = (lp->ls_3d.beg() + lp->ls_3d.direction() * (lp->ls_3d.beg().x() - crossing.x() * lp->ls_3d.beg().z()) / t1).lengthSquared();
761  else
762  l2 = (lp->ls_3d.beg() + lp->ls_3d.direction() * (lp->ls_3d.beg().y() - crossing.y() * lp->ls_3d.beg().z()) / t2).lengthSquared();
763 
764  if (l1 > l2)
765  above = std::move(lp);
766  else
767  under = std::move(lp);
768  }
769  else
770  under = std::move(lp);
771  }

874  void splitSort(std::unique_ptr<PolygonPrimitive> pp, std::list<std::unique_ptr<PolygonPrimitive>> &under, std::list<std::unique_ptr<PolygonPrimitive>> &above, float fl, float eps) override
875  {
876  bool p_on_front = false, intersects = false;
877  size_t i = 0;
878 
879  // find a point of pp more distant than eps and determine, on which side of this polygon it lies
880  for (; i < pp->poly_3d.points().size(); i++)
881  {
882  float d = poly_3d.normal().dot(pp->poly_3d.points()[i]) - poly_3d.distance();
883  if (std::abs(d) > eps)
884  {
885  p_on_front = d > 0;
886  break;
887  }
888  }
889 
890  if (i == pp->poly_3d.points().size()) // if none distant enough point was found, take the mean and decide
891  {
892  auto v_sum = Vector3f::zeros();
893  for (const auto &pt : pp->poly_3d.points())
894  v_sum += pt;
895  p_on_front = poly_3d.normal().dot(v_sum / pp->poly_3d.points().size()) - poly_3d.distance() > 0;
896  }
897  else // else search for intersections of planes
898  {
899  if (p_on_front)
900  {
901  for (i++; i < pp->poly_3d.points().size(); i++)
902  {
903  if (poly_3d.normal().dot(pp->poly_3d.points()[i]) - poly_3d.distance() < -eps)
904  {
905  intersects = true;
906  break;
907  }
908  }
909  }
910  else
911  {
912  for (i++; i < pp->poly_3d.points().size(); i++)
913  {
914  if (poly_3d.normal().dot(pp->poly_3d.points()[i]) - poly_3d.distance() > eps)
915  {
916  intersects = true;
917  break;
918  }
919  }
920  }
921  }
922 
923  if (!intersects)
924  {
925  if (p_on_front)
926  above.push_back(std::move(pp));
927  else
928  under.push_back(std::move(pp));

930  }

933  struct BrPt
934  {
935  BrPt() = default;
936  BrPt(const Vector3f& p3, const Vector2f &p2, bool above) : pt3d(p3), pt2d(p2), on_above(above) {}
937  BrPt(const Vector3f& p3, const Vector2f &p2, float il_t) : pt3d(p3), pt2d(p2), t(il_t) {}
938 
939  Vector3f pt3d;
940  Vector2f pt2d;
941  float t{std::numeric_limits<float>::quiet_NaN()};

943  bool from_under{false}, from_above{false}, to_under{false}, to_above{false};
944  int next_poly_under{0}, next_poly_above{0};
945

947  };
948 
949  std::list<BrPt> with_br_pts;
950  std::map<float, std::list<BrPt>::iterator> br_pts;
951  BrPt end_br_pt, tmp_br_pt;
952  Vector3f br_pt_first_3d = Vector3f::nan();
953  Vector3f il_d = poly_3d.normal().cross(pp->poly_3d.normal()).normalized();
954 
955  for (size_t j = pp->proj_2d.points().size() - 1, k = 0; k < pp->proj_2d.points().size(); j = k, k++)
956  {
957  // if a point of pp lies on *this, it lies on intersection line as well
958  float j_to_il = poly_3d.normal().dot(pp->poly_3d.points()[j]) + poly_3d.d();
959 
960  // if j-th point of pp lies on the intersection line, add it as a break point
961  if (std::abs(j_to_il) < eps)
962  {
963  float t = 0.0f;
964  if (br_pt_first_3d.hasNaN())
965  br_pt_first_3d = pp->poly_3d.points()[j];
966  else
967  t = (pp->poly_3d.points()[j] - br_pt_first_3d).dot(il_d);
968  with_br_pts.emplace_back(pp->poly_3d.points()[j], pp->proj_2d.points()[j], t);
969  br_pts[t] = --with_br_pts.end();
970  }
971  else
972  {
973  with_br_pts.emplace_back(pp->poly_3d.points()[j], pp->proj_2d.points()[j], j_to_il > 0);
974 
975  // if k-th point of pp is not on the intersection line, check if j-to-k line segment crosses il
976  if(std::abs(poly_3d.normal().dot(pp->poly_3d.points()[k]) + poly_3d.d()) > eps)
977  {
978  Vector3f tmp_v3d = pp->poly_3d.points()[k] - pp->poly_3d.points()[j];
979  float t_ls_p = -(poly_3d.normal().dot(pp->poly_3d.points()[j]) + poly_3d.d()) / poly_3d.normal().dot(tmp_v3d);
980 
981  if (t_ls_p > 0.0f && t_ls_p < 1.0f)
982  {
983  tmp_v3d = pp->poly_3d.points()[j] + tmp_v3d * t_ls_p;
984 
985  float t = 0.0f;
986  if (br_pt_first_3d.hasNaN())
987  br_pt_first_3d = tmp_v3d;
988  else
989  t = (tmp_v3d - br_pt_first_3d).dot(il_d);
990  with_br_pts.emplace_back(tmp_v3d, -Vector2f(tmp_v3d.x(), tmp_v3d.y()) * fl / tmp_v3d.z(), t);
991  br_pts[t] = --with_br_pts.end();
992  }
993  }
994  }
995  }
996 
997  for (auto k = std::prev(with_br_pts.end()), j = std::prev(k), l = with_br_pts.begin(); l != with_br_pts.end(); j = k, k = l, l++)
998  {
999  if (!j->isOnBrLine())
1000  {
1001  if (j->on_above)

1003  else
1004  k->from_under = true;
1005  }

1007  {

1010  else
1011  k->to_under = true;
1012  }
1013  }
1014

1016  for (const auto &bp : br_pts)
1017  {
1018  if (bp.second->to_above)
1019  npa++;
1020  if (bp.second->to_under)
1021  npu++;

1023  npa--;
1024  if (bp.second->from_under)
1025  npu--;
1026  bp.second->next_poly_under = npu;
1027  bp.second->next_poly_above = npa;
1028  }

1030  auto increment_circular = [&with_br_pts](std::list<BrPt>::iterator &it)
1031  {
1032  it++;
1033  if (it == with_br_pts.end())
1034  it = with_br_pts.begin();
1035  };
1036 
1037  for (const auto &bp : br_pts)
1038  {
1039  if (bp.second->to_above || bp.second->to_under)
1040  {
1041  std::unique_ptr<PolygonPrimitive> pp_new(new PolygonPrimitive(*pp, false));
1042  auto first_bp = bp.second;
1043  auto first_bp_to_above = first_bp->to_above;
1044  auto pt_to_add = first_bp;
1045  decltype(pt_to_add) previous_pt_to_add;
1046 
1047  do
1048  {
1049  pp_new->proj_2d.addPoint(pt_to_add->pt2d);
1050  pp_new->poly_3d.addPointDirect(pt_to_add->pt3d);
1051 
1052  if (pt_to_add->isOnBrLine())
1053  {
1054  if (pt_to_add->to_above && first_bp_to_above)
1055  {
1056  pt_to_add->to_above = false;
1057  previous_pt_to_add = pt_to_add;
1058  increment_circular(pt_to_add);
1059  }
1060  else if (pt_to_add->to_under && !first_bp_to_above)
1061  {
1062  pt_to_add->to_under = false;
1063  previous_pt_to_add = pt_to_add;
1064  increment_circular(pt_to_add);
1065  }
1066  else if (((pt_to_add->next_poly_above != 0 && first_bp_to_above) ||
1067  (pt_to_add->next_poly_under != 0 && !first_bp_to_above)) &&
1068  (previous_pt_to_add != (std::next(br_pts.find(pt_to_add->t)))->second))
1069  {
1070  previous_pt_to_add = pt_to_add;
1071  pt_to_add = (std::next(br_pts.find(pt_to_add->t)))->second;
1072  }
1073  else
1074  {
1075  previous_pt_to_add = pt_to_add;
1076  pt_to_add = (std::prev(br_pts.find(pt_to_add->t)))->second;
1077  }
1078  }
1079  else
1080  {
1081  previous_pt_to_add = pt_to_add;
1082  increment_circular(pt_to_add);
1083  }
1084  } while (pt_to_add != first_bp);
1085 
1086  if (first_bp_to_above)
1087  above.push_back(std::move(pp_new));
1088  else
1089  under.push_back(std::move(pp_new));
1090  }
1091  }
1092  }
