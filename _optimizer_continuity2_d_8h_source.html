75  bool operator()(const std::vector<VectorType> &pts, const SumArray &sum_array, std::vector<Approximation> &lines, std::vector<IndexType> &indices)
76  {
77  if (lines.size() < 2)
78  return true;
79  else
80  {
81  size_t i = 1;
82  while (i < lines.size())
83  {
84  if (!Approximation::getCrossing(lines[i - 1], lines[i], v_tmp) ||
85  VectorType::distanceSquared(v_tmp, (pts[indices[i - 1].second - 1] + pts[indices[i - 1].second]) / 2) > delta2)
86  {
87  if (indices[i].second - indices[i - 1].first < 6)
88  return false;
89  else
90  {
91  m1 = (2 * indices[i - 1].first + indices[i].second) / 3;
92  m2 = (indices[i - 1].first + 2 * indices[i].second) / 3;
93 
94  app1(sum_array.sums(indices[i - 1].first, m1));
95  app2(sum_array.sums(m1, m2));
96  app3(sum_array.sums(m2, indices[i].second));
97 
98  lines[i - 1] = app1;
99  indices[i - 1] = std::make_pair(indices[i - 1].first, m1);
100  lines[i] = app3;
101  indices[i] = std::make_pair(m2, indices[i].second);
102 
103  lines.insert(lines.begin() + i, app2);
104  indices.insert(indices.begin() + i, IndexType(m1, m2));
105  continue;
106  }
107  }
108  i++;
109  }
110  }
111  return true;
112  }
113 
114  private:
115  ElementType delta2;
116  VectorType v_tmp;
117  size_t m1{}, m2{};
118  Approximation app1, app2, app3;
119  };
